/**
 * Fill Resep form with payload data
 * Follows fill order cascade from Section 10.1
 */
export async function fillResepForm(payload: ResepFillPayload): Promise<FillResult> {
  const result: FillResult = {
    success: [],
    failed: [],
    skipped: [],
  };

  // STEP 1: Fill static fields (0ms)
  if (payload.static.no_resep) {
    const success = fillTextField('input[name="Resep[no_resep]"]', payload.static.no_resep);
    if (success) {
      result.success.push({ field: 'no_resep', value: payload.static.no_resep, method: 'direct' });
    } else {
      result.failed.push({ field: 'no_resep', error: 'Element not found' });
    }
  }

  // STEP 2: Fill AJAX display fields (500-1000ms wait)
  if (payload.ajax.ruangan) {
    try {
      const success = await fillAutocomplete(
        'input[name="ruangan"]',
        payload.ajax.ruangan,
        1000
      );
      if (success) {
        result.success.push({ field: 'ruangan', value: payload.ajax.ruangan, method: 'autocomplete' });
      } else {
        result.failed.push({ field: 'ruangan', error: 'Autocomplete timeout' });
      }
    } catch (error) {
      result.failed.push({ field: 'ruangan', error: String(error) });
    }
  }

  // STEP 3-9: Fill medication rows (follow Section 10.1 cascade)
  for (let i = 0; i < payload.medications.length; i++) {
    const med = payload.medications[i];
    
    // Fill racikan first (filters drugs)
    if (med.racikan) {
      const selector = `select[name="${RESEP_ROW_FIELDS.racikan(i)}"]`;
      const success = fillSelectField(selector, med.racikan);
      if (success) {
        result.success.push({ field: `medication[${i}].racikan`, value: med.racikan, method: 'direct' });
      }
      await sleep(200); // Wait for drug filter
    }

    // Fill obat_nama (AJAX, triggers stock fetch)
    if (med.nama_obat) {
      try {
        const selector = `input[name="${RESEP_ROW_FIELDS.obat_nama(i)}"]`;
        const success = await fillAutocomplete(selector, med.nama_obat, 1000);
        if (success) {
          result.success.push({ field: `medication[${i}].nama_obat`, value: med.nama_obat, method: 'autocomplete' });
        }
        await sleep(500); // Wait for stock fetch
      } catch (error) {
        result.failed.push({ field: `medication[${i}].nama_obat`, error: String(error) });
      }
    }

    // Continue with remaining fields...
    // jumlah, signa, aturan_pakai, keterangan
  }

  // STEP 10: Set prioritas (final field)
  if (payload.prioritas) {
    const success = fillSelectField('select[name="Resep[prioritas]"]', payload.prioritas);
    if (success) {
      result.success.push({ field: 'prioritas', value: payload.prioritas, method: 'direct' });
    }
  }

  console.log('[Resep] Fill complete:', result);
  return result;
}

## STEP 4: INTEGRATE WITH CONTENT SCRIPT

**Update**: `entrypoints/epuskesmas.content/index.ts`
```typescript
import { detectPage, getPelayananId } from './detector';
import { initResepPage, fillResepForm } from './page-resep';
import { sendMessage, onMessage } from '~/utils/messaging';

export default defineContentScript({
  matches: ['https://kotakediri.epuskesmas.id/*'],
  runAt: 'document_idle',
  main(ctx) {
    const pageType = detectPage(window.location.href);
    
    if (pageType === 'unknown') return;

    // Initialize page-specific handlers
    if (pageType === 'resep') {
      initResepPage();
    }

    // Listen for fill commands
    onMessage('execFill', async (cmd) => {
      if (cmd.pageType === 'resep') {
        return await fillResepForm(cmd.payload);
      }
      return { success: [], failed: [], skipped: [] };
    });

    // Report page ready
    sendMessage('pageReady', {
      pageType,
      pelayananId: getPelayananId(window.location.href),
      url: window.location.href,
    });
  },
});
```
